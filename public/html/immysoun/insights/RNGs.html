This page will show you my insights on RNGs.

this is an example of an HRNG written in C#, which uses the Cryptography API from System.Security:

      
      using System.Security.Cryptography;
      using System;
      
      class program
      {
          void Main()
          {
              //R.N.G Random Number Generator (RandInt(0, 100))
              RNGCryptoServiceProvider rng = new RNGCryptoServiceProvider();
              byte[] randomNumber = new byte[4];
              rng.GetBytes(randomNumber);
              int result = BitConverter.ToInt32(randomNumber,0);
              result = result % 101;
              if (result < 0)
                  result = -result;
          }
      }
